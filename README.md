/particle entity_effect{color:[255, 255, 255, 0]} ~ ~ ~ 0.0 0.0 0.0 1 1

/particle entity_effect{color:[255, 255, 245, 0]} ~ ~ ~ 0.0 0.0 0.0 1 1

/particle entity_effect{color:[255, 255, 235, 0]} ~ ~ ~ 0.0 0.0 0.0 1 1
